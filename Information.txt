<script>
function login() {
  var e = document.getElementById('email').value;
  if(e=="") { alert('Enter email'); }
  else {
    document.getElementById('login').className='center box hidden';
    document.getElementById('menu').className='center box';
  }
}
function logout() {
  document.getElementById('menu').className='center box hidden';
  document.getElementById('calc').className='center box hidden';
  document.getElementById('check').className='center box hidden';
  document.getElementById('login').className='center box';
  document.getElementById('email').value='';
  document.getElementById('pass').value='';
}
function showCalc(){ document.getElementById('menu').className='center box hidden'; document.getElementById('calc').className='center box'; }
function showCheck(){ document.getElementById('menu').className='center box hidden'; document.getElementById('check').className='center box'; }
function backFromCalc(){ document.getElementById('calc').className='center box hidden'; document.getElementById('menu').className='center box'; }
function backFromCheck(){ document.getElementById('check').className='center box hidden'; document.getElementById('menu').className='center box'; }

function gradeCalc(p){
  var g='F'; var r='Fail';
  if(p>=90){ g='A+'; r='Excellent'; }
  else if(p>=80){ g='A'; r='Very Good'; }
  else if(p>=70){ g='B'; r='Good'; }
  else if(p>=60){ g='C'; r='Average'; }
  else if(p>=50){ g='D'; r='Pass'; }
  else { g='F'; r='Fail'; }
  return [g,r];
}

function numFix(v){
  var n = Number(v);
  if(isNaN(n) || n<0) n=0;
  return n;
}

function generate(){
  var sname=document.getElementById('sname').value||'-';
  var sclass=document.getElementById('sclass').value||'-';
  var sroll=document.getElementById('sroll').value||'-';

  var chem=numFix(document.getElementById('chem').value); if(chem>75) chem=75;
  var comp=numFix(document.getElementById('compbio').value); if(comp>75) comp=75;
  var phys=numFix(document.getElementById('phys').value); if(phys>75) phys=75;
  var math=numFix(document.getElementById('math').value); if(math>75) math=75;
  var eng=numFix(document.getElementById('eng').value); if(eng>75) eng=75;
  var urdu=numFix(document.getElementById('urdu').value); if(urdu>75) urdu=75;
  var isl=numFix(document.getElementById('isl').value); if(isl>100) isl=100;
  var tar=numFix(document.getElementById('tar').value); if(tar>50) tar=50;

  var total=chem+comp+phys+math+eng+urdu+isl+tar;
  var max=75*6+100+50;
  var per=(total/max)*100;

  var pgchem=(chem/75)*100; var pgcomp=(comp/75)*100; var pgphys=(phys/75)*100;
  var pgmath=(math/75)*100; var pgend=(eng/75)*100; var pgurdu=(urdu/75)*100;
  var pgisl=(isl/100)*100; var pgtar=(tar/50)*100;

  var ggchem=gradeCalc(pgchem); var ggcomp=gradeCalc(pgcomp); var ggphys=gradeCalc(pgphys);
  var ggmath=gradeCalc(pgmath); var ggeng=gradeCalc(pgend); var ggurdu=gradeCalc(pgurdu);
  var ggisl=gradeCalc(pgisl); var ggtar=gradeCalc(pgtar);

  var gOver=gradeCalc(per);

  document.getElementById('nameOut').innerText='Name: '+sname;
  document.getElementById('classOut').innerText='Class: '+sclass;
  document.getElementById('rollOut').innerText='Roll No: '+sroll;
  document.getElementById('totalOut').innerText='Total Marks: '+total+' / '+max;
  document.getElementById('percOut').innerText='Percentage: '+per.toFixed(2)+'%';
  document.getElementById('gradeOut').innerText='Grade: '+gOver[0];
  document.getElementById('remOut').innerText='Remarks: '+gOver[1];

  document.getElementById('rchem').innerText=chem; document.getElementById('pchem').innerText=pgchem.toFixed(1)+'%';
  document.getElementById('gchem').innerText=ggchem[0]; document.getElementById('rcem').innerText=ggchem[1];

  document.getElementById('rcomp').innerText=comp; document.getElementById('pcomp').innerText=pgcomp.toFixed(1)+'%';
  document.getElementById('gcomp').innerText=ggcomp[0]; document.getElementById('rcompm').innerText=ggcomp[1];

  document.getElementById('rphys').innerText=phys; document.getElementById('pphys').innerText=pgphys.toFixed(1)+'%';
  document.getElementById('gphys').innerText=ggphys[0]; document.getElementById('rphysm').innerText=ggphys[1];

  document.getElementById('rmath').innerText=math; document.getElementById('pmath').innerText=pgmath.toFixed(1)+'%';
  document.getElementById('gmath').innerText=ggmath[0]; document.getElementById('rmathm').innerText=ggmath[1];

  document.getElementById('reng').innerText=eng; document.getElementById('peng').innerText=pgend.toFixed(1)+'%';
  document.getElementById('geng').innerText=ggeng[0]; document.getElementById('rengm').innerText=ggeng[1];

  document.getElementById('rurdu').innerText=urdu; document.getElementById('purdu').innerText=pgurdu.toFixed(1)+'%';
  document.getElementById('gurdu').innerText=ggurdu[0]; document.getElementById('rurdum').innerText=ggurdu[1];

  document.getElementById('risl').innerText=isl; document.getElementById('pisl').innerText=pgisl.toFixed(1)+'%';
  document.getElementById('gisl').innerText=ggisl[0]; document.getElementById('rislm').innerText=ggisl[1];

  document.getElementById('rtar').innerText=tar; document.getElementById('ptar').innerText=pgtar.toFixed(1)+'%';
  document.getElementById('gtar').innerText=ggtar[0]; document.getElementById('rtarm').innerText=ggtar[1];
}

function check(){
  var adm = document.getElementById('adm').value;
  var father = document.getElementById('dfather').value.toLowerCase();
  var outDiv = document.getElementById('checkRes');

  if(adm=="" || father==""){
    outDiv.innerHTML='<b>Enter admission and father name</b>';
    return;
  }

  fetch('students.json')
    .then(response => response.json())
    .then(data => {
      var found = false;
      for(var i=0;i<data.length;i++){
        var s = data[i];
        if(s.admission==adm && s.father.toLowerCase()==father){
          found = true;
          var total = s.math+s.physics+s.chemistry+s.english+s.urdu;
          var per = (total/500)*100;
          var html = '<h3>Result for '+s.name+'</h3>';
          html += 'Class: '+s.class+'<br>Admission No: '+s.admission+'<br>Father: '+s.father+'<br><br>';
          html += 'Math: '+s.math+'<br>Physics: '+s.physics+'<br>Chemistry: '+s.chemistry+'<br>English: '+s.english+'<br>Urdu: '+s.urdu+'<br>';
          html += 'Total: '+total+'/500<br>Percentage: '+per.toFixed(2)+'%<br>Grade: '+gradeCalc(per)[0];
          outDiv.innerHTML = html;
          break;
        }
      }
      if(!found) outDiv.innerHTML='<b>No student found.</b>';
    });
}
